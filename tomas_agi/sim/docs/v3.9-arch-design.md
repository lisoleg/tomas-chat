```python
# ── 跨模块统一类型 ──
from dataclasses import dataclass, field
from enum import Enum
from typing import Any, Dict, List, Optional, Tuple

class PsiAnchorLevel(Enum):
    """ψ-锚权限等级（从 gaussex_eml.PsiAnchorLevel 复制，作为跨模块基线）"""
    CONSTITUTIONAL = "constitutional"    # 宪法级 (I=1.0 Hard Veto)
    REGULATORY = "regulatory"            # 监管级 (0.8 <= I < 1.0)
    OPERATIONAL = "operational"          # 操作级 (I < 0.8)

class SnapResult(Enum):
    """κ-Snap 结果（从 ksnap_operator.SnapResult 引用）"""
    MANIFESTED = "manifested"
    REJECT_DZ = "reject_dz"
    SUSPEND_MUS = "suspend_mus"
    REJECT_FTEL = "reject_ftel"

@dataclass
class PsiAnchor:
    """ψ-锚数据结构（从 psi_anchor.PsiAnchor 引用）"""
    self_state: str
    kappa_at_write: int
    timestamp: str = field(default_factory=lambda: time.strftime("%Y-%m-%dT%H:%M:%S"))
    emotion_tone: Optional[str] = None
    continuation_branch: Optional[str] = None

@dataclass
class KSnapRecord:
    """κ-Snap 审计记录"""
    snap_id: str
    module: str
    result: str  # MANIFESTED / REJECT_DZ / SUSPEND_MUS
    i_value: float
    ftel_magnitude: float
    psi_anchor_id: str
    description: str
    timestamp: float = field(default_factory=time.time)
    snapshot_hash: str = ""

    def to_dict(self) -> Dict[str, Any]:
        return {
            "snap_id": self.snap_id,
            "module": self.module,
            "result": self.result,
            "i_value": self.i_value,
            "ftel_magnitude": self.ftel_magnitude,
            "psi_anchor_id": self.psi_anchor_id,
            "description": self.description,
            "timestamp": self.timestamp,
            "snapshot_hash": self.snapshot_hash,
        }

@dataclass
class MUSDualEntry:
    """MUS 双存条目"""
    entry_id: str
    description_a: str
    description_b: str
    code_a: str
    code_b: str
    created_at: float = field(default_factory=time.time)
    snap_ref: Optional[str] = None

    def to_dict(self) -> Dict[str, Any]:
        return {
            "entry_id": self.entry_id,
            "description_a": self.description_a,
            "description_b": self.description_b,
            "code_a": self.code_a,
            "code_b": self.code_b,
            "created_at": self.created_at,
            "snap_ref": self.snap_ref,
        }
```

### 详细任务分解

#### T1: Bug 修复 IDO i_value=0
- **文件**: `tomas_agi/sim/ido_bridge.py`
- **位置**: `IDOFlowTemplate.run_flow()` 方法, 当前行 273
- **变更**: `new_i = max(0.0, min(...))` → `new_i = max(1e-9, min(...))`
- **原因**: `random.uniform(-0.05, 0.05)` 噪声可使 `compute_I + noise ≤ 0` → i_value=0.0 导致后续除法/对数异常
- **验证**: 运行 `test_ido.py::test_simulate_tier1`

#### T2: Bug 修复 ARC API key 环境泄漏
- **文件**: `tomas_agi/sim/arc_api_client.py`
- **位置**: 第 53 行类构造函数, 第 351 行 CLI 入口
- **变更**: 
  - 行 53: `self.api_key = api_key or os.environ.get(...)` → `self.api_key = api_key if api_key is not None else os.environ.get(...)`
  - 行 351: 同上模式
- **原因**: `"" or os.environ.get(...)` — 空字符串是 falsy，导致回退到环境变量
- **验证**: 运行 `test_new_modules_v2.py::TestARCAPIClient::test_init_without_key`

### 7.6 管道模式

所有模块采用一致的管道执行模式：

```python
def pipeline_method(self, input_data, **kwargs) -> Dict[str, Any]:
    """
    五步管道 (模块间一致的执行顺序):
      1. 前置处理 (数据验证/类型检查)
      2. ψ锚扫描 (PsiAnchor.attach)
      3. 核心操作 (compress/build/schedule/generate)
      4. MUS 双存 (仅当存在互斥稳态)
      5. κ-Snap 审计 (记录结果)
    """
    result = {}
    # 1. 验证
    if not self._validate_input(input_data):
        raise ValueError(f"Invalid input: {type(input_data)}")
    # 2. ψ锚
    psi = PsiAnchor(self_state="...", kappa_at_write=self.kappa)
    result["psi_anchor"] = psi.to_dict()
    # 3. 核心
    output = self._core_operation(input_data, **kwargs)
    result["output"] = output
    # 4. MUS (条件)
    if self._has_conflict(output):
        result["mus"] = self._store_mus(output)
    # 5. κ-Snap
    result["audit"] = self._kappa_snap(output, psi)
    return result
```
